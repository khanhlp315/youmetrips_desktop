410  /**
411   * return the given content type or try to guess from stream or file name. Depending of the available data.
412   *
413   * @return the content type
414   */
415  public String getContentType() {
416    if (contentType != null) {
417      return contentType;
418    }
419    if (dataStream != null) {
420      try {
421        contentType = URLConnection.guessContentTypeFromStream(dataStream);
422      } catch (IOException ioe) {
423        // ignore exception
424      }
425    }
426    if (data != null) {
427      contentType = URLConnection.guessContentTypeFromName(filename);
428    }
429    // fallback - if we have no contenttype and cannot detect one, use 'application/octet-stream'
430    if (contentType == null) {
431      contentType = "application/octet-stream";
432    }
433    return contentType;
434  }
